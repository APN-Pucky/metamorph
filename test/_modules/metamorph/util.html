    result = ""
    codes = difflib.SequenceMatcher(a=old, b=new).get_opcodes()
    reds = []
    if on_color:
        for code in codes:
            if code[0] == "replace":
                reds += list(range(code[1],code[2]))
            elif code[0] == "delete":
                reds += list(range(code[1],code[2]))
